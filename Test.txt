def issueType = getFieldById("issuetype")?.value as String
def descField = getFieldById("description")

if (!descField.getValue()) {
    switch (issueType) {
        case "Bug":
            descField.setFormValue("Bug-Template …")
            break
        case "Story":
            descField.setFormValue("Story-Template …")
            break
        default:
            descField.setFormValue("")
    }
}
