Map<String, Integer> activeUserCountMap = new HashMap<>();
Map<String, Integer> inactiveUserCountMap = new HashMap<>();

groupUserMapping.put(group.getName(), usersInGroup);
kmUsers.addAll(usersInGroup);

// aktive/inaktive z√§hlen
int active = 0;
int inactive = 0;
for (ApplicationUser user : usersInGroup) {
    if (user.isActive()) {
        active++;
    } else {
        inactive++;
    }
}
activeUserCountMap.put(group.getName(), active);
inactiveUserCountMap.put(group.getName(), inactive);
