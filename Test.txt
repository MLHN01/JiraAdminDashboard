import com.onresolve.jira.groovy.user.FieldBehaviours
import static com.atlassian.jira.issue.IssueFieldConstants.*
import groovy.transform.BaseScript

@BaseScript FieldBehaviours fieldBehaviours

def descField = getFieldById(DESCRIPTION)
def issueType = getFieldById("issuetype")?.value as String

switch (issueType) {
    case "Bug":
        descField.setFormValue("""h3. Beschreibung des Fehlers

*Schritte zur Reproduktion:*
# 
# 
# 

*Erwartetes Ergebnis:*

*Tatsächliches Ergebnis:*""")
        break

    case "Story":
        descField.setFormValue("""h3. Benutzerstory

Als *[Rolle]* möchte ich *[Funktion]*, um *[Ziel]* zu erreichen.

*Akzeptanzkriterien:*
# 
# 
# """)
        break

    case "Task":
        descField.setFormValue("""h3. Aufgabentext

*Ziel der Aufgabe:*

*Erforderliche Schritte:*
# 
# 
# """)
        break

    default:
        descField.setFormValue("") // leer lassen für andere
        break
}
